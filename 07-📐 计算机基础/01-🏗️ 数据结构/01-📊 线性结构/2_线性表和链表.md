## 1 线性表的定义与特点

定义：由 n（ n ⩾0 ）个数据特性相同的元素构成的有限序列，称为线性表，线性表是 n 个数据元素的有限序列，其中 n 个数据是相同数据类型的

线性表中元素的个数 n（ ）定义为线性表的长度，当 n=0 时称之为空表。 对于非空的线性表或线性结构，其特点是：
- 存在唯一的一个被称作“第一个”的数据元素； 
- 存在唯一的一个被称作“最后一个”的数据元素； 
- 除第一个元素外，结构中的每个数据元素均只有一个前驱； 
- 除最后一个元素外，结构中的每个数据元素均只有一个后继。

## 3 顺序表

**顺序表的定义**：用一组`连续的内存单元`依次`存储线性表的各个元素`，也就是说，逻辑上相邻 的元素，实际的物理存储空间也是连续的。

### 3.4 插入

尾插法（从后往前），注意脚标越界

```c
Status SqlInsert(Sqlist *L, int pos, SqlElemType e) {  
    // 判断边界条件  
    if (pos < 0 || pos > L->length) {  
        return ERROR;  
    }  
    if (L->length > SQLMAXSIZE) {  
        return OVERFLOW;  
    }  
    // 当前顺序表的有效长度  
    int len = L->length;  
    // 从后往前进行插入  
    for (int i = len - 1; i >= pos; i--) {  
        L->base[i + 1] = L->base[i];  
    }  
    L->base[pos] = e;  
    L->length++;  
    return OK;  
}
```

改进点：
- 顺序表中的数组使用SQLMAXSIZE，固定了大小，后期添加阈值，使用realloc函数进行动态扩容
- 对于尾插可以直接优化，检查时间消耗
### 3.5 删除

**通过索引位置删除**
```c
Status SqlDeleteByIndex(Sqlist *L, int pos, SqlElemType *e) {  
    if (pos < 0 || pos >= L->length) {  
        return ERROR;  
    }  
    *e = *(L->base + pos);  
    int len = L->length;  
    for (int i = pos + 1; i < len; i++) {  
        // i从pos+1开始，防止越界  
        // L->base[i + 1] = L->base[i];  
        L->base[i - 1] = L->base[i];  
    }  
    L->length--;  
    return OK;  
}
```

**通过元素删除**
```c
Status SqlDeleteByElem(Sqlist *L,SqlElemType e){  
    int index = LocateElem(L,e);  
    SqlElemType res;  
    return SqlDeleteByIndex(L,index,&res);  
}
```

## 4 单链表（Single Linked List）

### 4.1 存储结构

- 单链表由头节点(不存放数据只存放下个节点的地址)和n个节点组成
- 每个节点分为两个域：数据域和指针域(存放下个节点的地址)
- 第n个节点的指针域为NULL

如下图所示

### 4.2 存储结构

```c
typedef int LlElemtype
typedef struct __LNode{
	LlElemtype data;   //存放单个节点的数据
	struct __LNode *next;     //存放下个节点的地址
}LNode,*LinkedList;
```
### 4.3 初始化和遍历

```c
// 单链表初始化  
Status InitLl(Linkedlist *L) {     //L是一个二级指针  
    (*L) = (Linkedlist) malloc(sizeof(LNode));  
    (*L)->next = NULL;  
    return OK;  
}

// 单链表从前到后遍历  
void Traverse(Linkedlist L){  
    if(!L->next){  
        printf("链表中不存在元素");  
        return;  
    }  
    Linkedlist p = L->next;  
    while(p){  
        printf("%d ",p->data);  
        p = p->next;  
    }  
}
```
### 4.4 创建链表（头插法）$O(n)$

```c
// 创建链表（头插法）  
// 注意：逆序存储，遍历需要反过来  
// 例如：输如 1 2 3 ，链表：头节点 -> 3 -> 2 -> 1 -> NULL// 一般我们忽略头结点是啥，当做一个起始位置即可  
void CreateLL_H(Linkedlist L, int n) {  
    printf("请输入 %d 个整数，作为链表的元素：", n);  
    fflush(stdout);  
    for (int i = 0; i < n; i++) {  
        Linkedlist p = (Linkedlist) malloc(sizeof(LNode));  
        int data;  
        scanf_s(" %d", &data);  
        p->data = data;  
        p->next = L->next;  
        L->next = p;  
    }  
}
```

### 4.5 创建链表（尾插法）$O(n)$

```c
// 创建链表（尾插法）  
void CreateLL_R(Linkedlist L, int n) {  
    printf("请输入 %d 个整数，作为链表的元素：", n);  
    fflush(stdout);  
    // 头结点保持不动，使用移动节点  
    Linkedlist pm = L;  
    for (int i = 0; i < n; i++) {  
        Linkedlist p = (Linkedlist) malloc(sizeof(LNode));  
        int data;  
        scanf_s(" %d", &data);  
        // 新节点赋值  
        p->data = data;  
        // 新节点尾部必然是NULL  
        p->next = NULL;  
        // 新节点插入到前一个节点后  
        pm -> next = p;  
        // 指针移动到新节点的前一个节点  
        pm = pm -> next;  
    }  
}
```

### 4.8 插入元素$O(n)$📕

```c
//插入元素  
Status Insert(Linkedlist L, int pos, LlElemtype e) {  
    if(pos<1){  
        return ERROR;  
    }  
    //确定插入的位置，即pos前1位  
    Linkedlist p = L;  
    // 注意p从0开始，即L开始，所以最后的位置是pos的前1个节点  
    for (int i = 0; i < pos - 1; i++) {  
        p = p->next;  
        if(!p){  
            printf("pos超过链表长度\n");  
            return ERROR;  
        }  
    }  
    // 判断获得的节点是否合法  
    //创建新节点  
    Linkedlist pnew = (Linkedlist) malloc(sizeof(LNode));  
    pnew->data = e;  
    // 此时p为pos的前1个节点，开始替换  
    pnew->next = p->next;  
    p->next = pnew;  
    return OK;  
}
```
### 4.9 删除元素$O(n)$📕

```c
//删除元素  
Status Delete(Linkedlist L, int pos, LlElemtype *e) {  
    if(pos<1){  
        return ERROR;  
    }  
    //确定插入的位置，即pos前1位  
    Linkedlist p = L;  
    // 注意p最后的位置，pos的前1个节点  
    for (int i = 0; i < pos - 1; i++) {  
        p = p->next;  
        if(!p){  
            printf("pos超过链表长度\n");  
            return ERROR;  
        }  
    }  
    //获取删除的元素  
    Linkedlist dp = p->next;  
    *e = dp->data;  
    // 删除元素  
    p->next = p->next->next;  
    free(dp);  
    return OK;  
}
```

- 由此，从表中任意节点出发，可以找到其他节点

## 6 双向链表

### 6.4 插入和删除

```c
// 插入（从前往后）  
void Insert(DouLinkList D, int pos, DouLElemtype e) {  
    if (pos < 1) {  
        printf("输入位置有误\n");  
        return;  
    }  
    // 找到pos前一个元素  
    DouLinkList p = D;  
    for (int i = 0; i < pos - 1; i++) {  
        p = p->next;  
    }  
    // 交换连接关系（建议画图）  
    DouLinkList pnew = (DouLinkList) malloc(sizeof(DouLinkNode));  
    pnew->data = e;  
    // 前一个元素要换next，后一个元素要换prev  
    // 新增的元素要换next和prev  
    // 先换后一个元素的  
    pnew->next = p->next;  
    p->next->prev = pnew;  
    // 在换前一个元素的  
    p->next = pnew;  
    pnew->prev = p;  
}  
  
//删除  
void Delete(DouLinkList D, int pos, DouLElemtype *e){  
    if (pos < 1) {  
        printf("输入位置有误\n");  
        return;  
    }  
    // 找到pos前一个元素  
    DouLinkList p = D;  
    for (int i = 0; i < pos - 1; i++) {  
        p = p->next;  
    }  
    // 先获取要删除的节点  
    DouLinkList pfree = p->next;  
    *e = pfree->data;  
    // 交换连接关系  
    p->next = pfree->next;  
    pfree->next->prev = p;  
    free(pfree);  
    pfree = NULL;  
}
```

### 7.2 有序表合并（并归排序的基础）

```c
void MergeList_Seq(Sqlist *La, Sqlist *Lb, Sqlist *Lc) {  
    //1. 定义变量  
    // 获取长度  
    Lc->length = La->length + Lb->length;  
    // 获取a和b的首位节点  
    SqlElemType *pa = La->base, *pa_last = La->base + (La->length - 1);  
    SqlElemType *pb = Lb->base, *pb_last = Lb->base + (Lb->length - 1);  
    // 获取c的首节点  
    SqlElemType *pc = Lc->base;  
  
    //2.先比较a和b，将内容放入c中  
    while (pa <= pa_last && pb <= pb_last) {  
        if (*pa < *pb) {  
            *(pc++) = *(pa++);  
        } else {  
            *(pc++) = *(pb++);  
        }  
    }  
  
    // 3. 剩下多余的继续填充  
    while (pa <= pa_last) {  
        *(pc++) = *(pa++);  
    }  
  
    while (pb <= pb_last) {  
        *(pc++) = *(pb++);  
    }  
}
```
### 7.3 有序链表合并

```c
void Merge_LinkedList(Linkedlist *La, Linkedlist *Lb, Linkedlist *Lc) {  
    //1。定义变量  
    // 先让Lc指向La  
    *Lc = *La;  
    // 获取a和b的元素首节点  
    Linkedlist pa = (*La)->next;  
    Linkedlist pb = (*Lb)->next;  
    // 获取c的元素首节点  
    Linkedlist pc = *Lc;  
  
    //2.有序链接a和b的元素  
    while (pa && pb) {  
        if (pa->data < pb->data) {  
            pc->next = pa;  
            //移动pc指针  
            pc = pa;  
            //移动pa指针  
            pa = pa->next;  
        } else {  
            pc->next = pb;  
            pc = pb;  
            pb = pb->next;  
        }  
    }  
  
    //3.还有多的元素  
    pc->next = pa ? pa : pb;  
    free(*Lb);  
    *La = * Lb = NULL;  
}
```
